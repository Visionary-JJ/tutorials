IE562: Assignment 2¶

Use BFS to solve the 3M3C problem. The DFS solution is given below.

Problem Description¶

3 missionaries and 3 cannibals are at the one side of a river

There is a boat on the river

Objective:¶

Move all 3 missionaries and 3 cannibals to the another side of the river safely

Constraints:¶

  1. At most two people can be on the boat
  2. Need at least one people on the boat to move from one side to another side
  3. Whenever the number of canibals are larger than the number of missionaries (on the same side), the missionaries will be aten by canibals, and the task is failed

$S=[s_1,s_2,s_3,s_4,s_5]$

$s_1$: the number of missionaries on the left side of the river

$s_2$: the number of cannibals on the left side of the river

$s_3$: the location of the boat, if it is 1/0 then it is on the left/right side

$s_4$: the number of missionaries on the right side of the river

$s_5$: the number of cannibals on the right side of the river

Initial State:¶

$ S_0=[3,3,1,0,0] $

Goal State:¶

$ S^*=[0,0,0,3,3] $

Action Space and State Transition:¶

$S=[s_1,s_2,s_3,s_4,s_5]$

  1. if $s_3=1$:

    $0\le \delta_{s_i}\le \min\{2,s_i\},i=1,2$

    $\delta_{s_1}+\delta_{s_2}\le2$

    if $\delta_{s_1}+\delta_{s_2}\ge 1$:

    $S_{new}=[s_1-\delta_{s_{1}},s_2-\delta_{s_{2}},0,s_4+\delta_{s_{1}},s_5+\delta_{s_{2}}]$

    else:

    $S_{new}=S$

  1. if $s_3=0$:

    $0\le \delta_{s_i}\le \min\{2,s_i\},i=4,5$

    $\delta_{s_4}+\delta_{s_5}\le2$

    if $\delta_{s_4}+\delta_{s_5}\ge 1$:

    $S_{new}=[s_1+\delta_{s_{4}},s_2+\delta_{s_{5}},1,s_4-\delta_{s_{4}},s_5-\delta_{s_{5}}]$

    else:

    $S_{new}=S$

Stop¶

$S=[s_1,s_2,s_3,s_4,s_5]$

if $s_1<s_2$ or $s_4<s_5$ then stop (failed)

if $s_1=s_2=0$ and $s_4=s_5=3$ then stop (goal state)

def obtain_actions(S):
    s3=S[2]
    actions=[]
    if s3==1:
        s1=S[0]
        s2=S[1]
        # delta1 in [0,s1]
        # delta2 in [0,s2]
        # delta1+delta2<=2
        for delta1 in range(s1+1):
            for delta2 in range(s2+1):
                if delta1+delta2<=2 and delta1+delta2>0:
                    actions.append([delta1,delta2])
    else:
        s4=S[3]
        s5=S[4]
        for delta4 in range(s4+1):
            for delta5 in range(s5+1):
                if delta4+delta5<=2 and delta4+delta5>0:
                    actions.append([delta4,delta5])
    return actions

def failure(S):
    if (S[1]>S[0] and S[0]>0) or (S[4]>S[3] and S[3]>0):
        failed=True
    else:
        failed=False
    return failed

def next_state(S,action):
    s3=S[2]
    S_new=S[:]
    if s3==1:
        S_new[0]-=action[0]
        S_new[1]-=action[1]
        S_new[2]=0
        S_new[3]+=action[0]
        S_new[4]+=action[1]
    else:
        S_new[0]+=action[0]
        S_new[1]+=action[1]
        S_new[2]=1
        S_new[3]-=action[0]
        S_new[4]-=action[1]
    failed=failure(S_new)
    return S_new,failed

def DFS(S,V,graph,goal_state=[0,0,0,3,3]):
    win=False
    if str(S) not in graph:
        graph[str(S)]=[]
        
    if S==goal_state:
        win=True
        print("Reached the goal state!")
        return V,win,graph
    
    print("Current State:",S)
    actions=obtain_actions(S)
    for action in actions:
        S_new,failed=next_state(S,action)
        print("New state:{}|failed:{}|visited:{}".format(S_new,failed,str(S_new) in V))
        if str(S_new) not in V:
            V.append(str(S_new))
            graph[str(S)].append(str(S_new))
            if not failed:
                print("    Exploring")
                V,win,graph=DFS(S_new,V,graph)
                if win:
                    break
    return V,win,graph

def BFS(S,visit,stack,graph,goal_state=[0,0,0,3,3]):
    win=False
    print("exploring {}".format(S))
    stack.remove(str(S))
    visit.append(str(S))
    if str(S)==str(goal_state):
        win=True
        return visit,stack,graph,win
    
    if str(S) not in graph:
        graph[str(S)]=[]
        
    actions=obtain_actions(S)        
    for action in actions:
        S_new,failed=next_state(S,action)
        graph[str(S)].append(str(S_new))
        
        if str(S_new) not in visit:
            print("    new child of {}:{} (failed:{})".format(S,S_new,failed))
            if not failed:
                stack.append(str(S_new))
        else:
            print("    visited child {}".format(S_new))
    for Sp in stack:
        Sp=[int(s) for s in Sp[1:-1].replace(" ","").split(",")]
        if str(Sp) not in visit:
            visit,stack,graph,win=BFS(Sp,visit,stack,graph)
        if win:
            break
    return visit,stack,graph,win

def BFS(root,goal_state=[0,0,0,3,3]):
    visiteed, queue = [str([0,0,0,3,3])],collections.deque([root])
    
    graph1 = {}
    while queue:
        vertex = queue.popleft()
        
        
        if vertex==goal_state:
            print("goal reached sucessfully",vertex)
            return visiteed, queue,graph1
            break
        
        if str(vertex) not in visiteed:
            
            graph1[str(vertex)] = []
            try:
                
                for action in obtain_actions(vertex):
                    (a,b)=next_state(vertex,action)
                    if str(a) not in visiteed:
                        if b==False:
                            queue.append(a)
                            print("states in queue:",queue)
                            graph1[str(vertex)].append(str(a))
                        else:
                            visiteed.append(str(a))
                        
                
                
                
                
            except:
                pass
            visiteed.append(str(vertex))
            print("visited states:",visiteed)
    
    return visiteed,queue,graph1
